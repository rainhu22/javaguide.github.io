  4. class Solution {
    // int max = -1;
    // int index=0;
    public TreeNode constructMaximumBinaryTree(int[] nums) {
        return buildTree(0,nums.length-1,nums);
    }
    public TreeNode buildTree(int left, int right,int[] nums){
        int max = -1;
        int index=0;
        if(left>right) return null;
        for(int i=left;i<=right;i++){
            if(nums[i]>max){
                max=nums[i];
                index = i;}
            }
        TreeNode node = new TreeNode(max);
        node.left = buildTree(left,index-1, nums);  // *8
        node.right = buildTree(index+1,right, nums);
        return node;
    }
}

/*为何不能在函数外定义max和index？ 反正每次进入函数都会赋值
进入函数并未初始化，node.left将index改变后传入right的index不再是本层的index，已经被前一行(*8)函数改变
而所需要的结果是在一个buildTree内部index只被改变一次
*/
